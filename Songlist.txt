        <!-- Song selection for each Sunday will be added here -->
        <!-- You'll need JavaScript to dynamically add options based on the month -->
        <button type="submit">Submit</button>
    </form>

    <script>
        // Sample song options
        const songOptions = ['Song 1', 'Song 2', 'Song 3', 'Song 4', 'Song 5'];

        // Function to dynamically add song selection options
        function addSongSelectionOptions() {
            const form = document.getElementById('songForm');
            const daysInMonth = new Date(new Date().getFullYear(), new Date().getMonth() + 1, 0).getDate(); // Get number of days in current month

            for (let i = 1; i <= daysInMonth; i++) {
                if (new Date().getDay(new Date(new Date().getFullYear(), new Date().getMonth(), i)) === 0) { // Check if it's a Sunday
                    const label = document.createElement('label');
                    label.textContent = `Select song for Sunday, ${i}.`;

                    const select = document.createElement('select');
                    songOptions.forEach(song => {
                        const option = document.createElement('option');
                        option.value = song;
                        option.textContent = song;
                        select.appendChild(option);
                    });

                    form.appendChild(label);
                    form.appendChild(select);
                }
            }
        }

        // Call the function to add song selection options when the page loads
        window.addEventListener('DOMContentLoaded', addSongSelectionOptions);
    </script>
